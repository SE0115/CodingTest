let input = require("fs")
  .readFileSync("/dev/stdin")
  .toString()
  .trim()
  .split("\n");

class Queue {
  constructor() {
    this.item = {};
    this.head = 1;
    this.tail = 1;
  }
  enqueue(item) {
    this.item[this.tail] = item;
    this.tail++;
  }
  dequeue() {
    if (this.tail === this.head) return undefined;
    const item = this.item[this.head];
    delete this.item[this.head];
    this.head += 1;
    return item;
  }
  size() {
    return this.tail - this.head;
  }
}

const [cond, ...info] = input;
const [N, M, K, X] = cond.split(" ").map(Number);
const distance = Array(N + 1).fill(-1);
const graph = Array.from(Array(N + 1), () => []);

for (let i of info) {
  const [x, y] = i.split(" ").map(Number);
  graph[x].push(y);
}

distance[X] = 0;
bfs(X);

function bfs(start) {
  const queue = new Queue();
  queue.enqueue(start);

  while (queue.size()) {
    const now = queue.dequeue();
    for (let next of graph[now]) {
      if (distance[next] !== -1) continue;
      distance[next] = distance[now] + 1;
      queue.enqueue(next);
    }
  }
}

const result = [];
distance.map((x, idx) => {
  if (x === K) result.push(idx);
});

console.log(result.length ? result.join("\n") : -1);
