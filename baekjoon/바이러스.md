let input = require("fs")
  .readFileSync("/dev/stdin")
  .toString()
  .trim()
  .split("\n");

class Queue {
  constructor() {
    this.item = {};
    this.head = 1;
    this.tail = 1;
  }
  enqueue(item) {
    this.item[this.tail] = item;
    this.tail++;
  }
  dequeue() {
    if (this.tail === this.head) return undefined;
    const item = this.item[this.head];
    delete this.item[this.head];
    this.head += 1;
    return item;
  }
  size() {
    return this.tail - this.head;
  }
}

function bfs(start) {
  let queue = new Queue();
  queue.enqueue(start);

  while (queue.size()) {
    let node = queue.dequeue();
    if (!visited[node]) {
      cnt++;
      visited[node] = true;
    }
    for (i of graph[node]) {
      if (!visited[i]) {
        queue.enqueue(i);
      }
    }
  }
}

let [computer, num, ...connects] = input;
let graph = new Array(Number(computer) + 1).fill([]);
let visited = Array(Number(computer) + 1).fill(false);

for (let connect of connects) {
  const [x, y] = connect.split(" ").map(Number);
  graph[x] = [...graph[x], y];
  graph[y] = [...graph[y], x];
}

let cnt = 0;

bfs(1);
console.log(cnt - 1);
