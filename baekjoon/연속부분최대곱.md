```js
let input = require("fs")
  .readFileSync("/dev/stdin")
  .toString()
  .trim()
  .split("\n")
  .map((x) => Number(x));

let [size, ...nums] = input;

for (let i = 0; i < size; i++) {
  if (i) {
    nums[i] = Math.max(nums[i], nums[i - 1] * nums[i]);
  }
}

console.log(Math.max(...nums).toFixed(3));
```
