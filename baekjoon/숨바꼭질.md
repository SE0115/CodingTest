let input = require("fs")
  .readFileSync("/dev/stdin")
  .toString()
  .trim()
  .split("\n");

class Queue {
  constructor() {
    this.item = {};
    this.head = 1;
    this.tail = 1;
  }
  enqueue(item) {
    this.item[this.tail] = item;
    this.tail++;
  }
  dequeue() {
    if (this.tail === this.head) return undefined;
    const item = this.item[this.head];
    delete this.item[this.head];
    this.head += 1;
    return item;
  }
  size() {
    return this.tail - this.head;
  }
}

let [curPos, target] = input[0].split(" ").map(Number);
const MAX = 100000;
let visited = new Array(MAX + 1).fill(0);

function bfs() {
  let queue = new Queue();
  queue.enqueue(curPos);
  while (queue.size()) {
    let cur = queue.dequeue();
    if (cur === target) return visited[cur];

    for (let next of [cur - 1, cur + 1, cur * 2]) {
      if (next < 0 || next > MAX) continue;
      if (visited[next] === 0) {
        visited[next] = visited[cur] + 1;
        queue.enqueue(next);
      }
    }
  }
}

console.log(bfs());
