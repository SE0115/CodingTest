```js
let input = require("fs")
  .readFileSync("/dev/stdin")
  .toString()
  .trim()
  .split("\n");

const [cond, seq] = input;
let freq = new Map();

for (let i of seq.split(" ")) {
  if (freq.has(i)) {
    freq.set(i, freq.get(i) + 1);
  } else {
    freq.set(i, 1);
  }
}

const sorted = [...freq.entries()].sort((a, b) => b[1] - a[1]);
let result = "";

for (let [key, cnt] of sorted) {
  for (let i = 0; i < cnt; i++) {
    result += key + " ";
  }
}

console.log(result);
```
