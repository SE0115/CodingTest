```js
let input = `4
9 5 4 8`.split('\n')

const nums = input[1].split(' ')
let result = []
let tmp = -1;
const length = Number(input[0])
for(let i=0; i<length-1; i++) {
  let pivot = nums[i]
  for(let j=i+1; j<length; j++) {
    if(nums[j]>pivot) {
      tmp = Number(nums[j])
      break
    }
  }
  result.push(tmp)
  tmp=-1
}
result.push(-1)
console.log(result.join(' '))
```
