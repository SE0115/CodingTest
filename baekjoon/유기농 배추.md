let input = require("fs")
  .readFileSync("/dev/stdin")
  .toString()
  .trim()
  .split("\n");

class Queue {
  constructor() {
    this.item = {};
    this.head = 1;
    this.tail = 1;
  }
  enqueue(item) {
    this.item[this.tail] = item;
    this.tail++;
  }
  dequeue() {
    if (this.tail === this.head) return undefined;
    const item = this.item[this.head];
    delete this.item[this.head];
    this.head += 1;
    return item;
  }
  size() {
    return this.tail - this.head;
  }
}

const dx = [-1, 1, 0, 0];
const dy = [0, 0, -1, 1];

let [num, ...testcase] = input;
let idx = 0;

for (let i = 0; i < num; i++) {
  const [M, N, l_num] = testcase[idx++].split(" ").map(Number);
  let cnt = 0;

  graph = [...Array(N)].map((e) => Array(M).fill(0));

  for (let j = 0; j < l_num; j++) {
    const [x, y] = testcase[idx++].split(" ").map(Number);
    graph[y][x] = 1;
  }

  for (let n = 0; n < N; n++) {
    for (let m = 0; m < M; m++) {
      if (graph[n][m]) {
        bfs(n, m);
        cnt++;
      }
    }
  }

  function bfs(x, y) {
    const queue = new Queue();
    queue.enqueue([x, y]);

    while (queue.size()) {
      const [x, y] = queue.dequeue();
      if (!graph[x][y]) continue;
      graph[x][y] = 0;

      for (let i = 0; i < 4; i++) {
        const nx = x + dx[i];
        const ny = y + dy[i];

        if (nx < 0 || nx >= N || ny < 0 || ny >= M) continue;
        if (graph[nx][ny]) {
          queue.enqueue([nx, ny]);
        }
      }
    }
  }
  console.log(cnt);
}
