function DFS(x, y) {
  if (x <= -1 || x >= aparts || y <= -1 || y >= aparts) return 0;
  if (graph[x][y] >= 1) {
    // ì•„ì§ ë°©ë¬¸í•˜ì§€ ì•Šì€ ê²½ìš°
    graph[x][y] = -1;

    // 4ë°©í–¥ í˜¸ì¶œ
    result += DFS(x - 1, y);
    result += DFS(x, y - 1);
    result += DFS(x + 1, y);
    result += DFS(x, y + 1);
    return result;
  }
  return 0;
}

let input = require("fs")

  graph.push(i.split("").map((x) => Number(x)));
}
