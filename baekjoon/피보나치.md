let input = require("fs")
  .readFileSync("/dev/stdin")
  .toString()
  .trim()
  .split("\n");

let [num, ...arr] = input;

let fibo = [];
fibo.push(0);
fibo.push(1);

while (fibo[fibo.length - 1] < 1e9)
  fibo.push(fibo[fibo.length - 2] + fibo[fibo.length - 1]);

for (let i of arr) {
  let idx = fibo.length - 1;
  let result = [];
  while (i > 0) {
    if (i >= fibo[idx]) {
      i -= fibo[idx];
      result.push(fibo[idx]);
    }
    idx--;
  }
  console.log(result.sort((a, b) => a - b).join(" "));
}
