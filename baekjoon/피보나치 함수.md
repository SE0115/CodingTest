const input =  require('fs').readFileSync('/dev/stdin').toString().trim().split('\n').map(x => Number(x));

const len = input.shift();

for(let i=0;i<len;i++){
    const fibo = [[1,0],[0,1]];
    for(let j=2; j<=input[i]; j++){
        fibo[j] = [
            fibo[j-1][0] + fibo[j-2][0],
            fibo[j-1][1] + fibo[j-2][1]
        ];
    }
    console.log(fibo[input[i]].join(' '));
}
