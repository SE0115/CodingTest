```js
let input = require("fs")
  .readFileSync("/dev/stdin")
  .toString()
  .trim()
  .split("\n");

let [num, weight] = input;
weight = weight
  .split(" ")
  .map((x) => Number(x))
  .sort((a, b) => a - b);

let result = 1;
for (x of weight) {
  if (result < x) {
    break;
  } else {
    result += x;
  }
}

console.log(result);
```
