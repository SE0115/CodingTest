let input = require('fs').readFileSync('/dev/stdin').toString().trim().split('\n').map(x => Number(x))

const [num, ...straws] = input

straws.sort((a,b) => b-a)
let maxTri = -1;
for(let i=0; i<num-2; i++) {
  if(straws[i] < straws[i+1] + straws[i+2]) {
      const total = straws[i] + straws[i+1] + straws[i+2]
      if(maxTri < total) {
          maxTri = total
      }
  }
}
console.log(maxTri)
