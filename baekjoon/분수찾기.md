function searchFractions(order) {
  let num = 1;
  let sum = 0;
  
  while(sum<order) {
    sum += num++;
  }
  sum -= (num-1) 
  
  let cnt = 0;
  for(let i=1;i<num;i++) {
    if(++cnt === order-sum){
      if((num-1)%2 === 0) {
        console.log(`${i}/${num-i}`)
      }
      else {
        console.log(`${num-i}/${i}`)
      }
    }
  }
}
