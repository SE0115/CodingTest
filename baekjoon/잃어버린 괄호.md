let input = require('fs').readFileSync('/dev/stdin').toString().trim().split('-')

let result = 0
for(let i=0; i<input.length; i++) {
  let tmp = input[i].split('+').map(x=>Number(x)).reduce((acc, cur) => acc+cur)

  if(i === 0) {
    result += tmp
  }
  else {
    result -= tmp
  }
}

console.log(result)
