```js
let input = require("fs")
  .readFileSync("/dev/stdin")
  .toString()
  .trim()
  .split("\n");

const [num, need] = input[0].split(" ").map((x) => Number(x));
const trees = input[1]
  .split(" ")
  .map((x) => Number(x))
  .sort((a, b) => b - a);

let start = 0;
let end = trees[0];
let result = 0;

while (start <= end) {
  let mid = parseInt((start + end) / 2);
  let sum = 0;
  for (let tree of trees) {
    if (tree > mid) sum += tree - mid;
  }
  if (sum < need) {
    end = mid - 1;
  } else {
    result = mid;
    start = mid + 1;
  }
}

console.log(result);
```
