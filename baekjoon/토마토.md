let input = require("fs")
  .readFileSync("/dev/stdin")
  .toString()
  .trim()
  .split("\n");

class Queue {
  constructor() {
    this.item = {};
    this.head = 1;
    this.tail = 1;
  }
  enqueue(item) {
    this.item[this.tail] = item;
    this.tail++;
  }
  dequeue() {
    if (this.tail === this.head) return undefined;
    const item = this.item[this.head];
    delete this.item[this.head];
    this.head += 1;
    return item;
  }
  size() {
    return this.tail - this.head;
  }
}

const dx = [-1, 1, 0, 0];
const dy = [0, 0, -1, 1];

const [cond, ...fields] = input;
const [w, h] = cond.split(" ").map(Number);
const field = fields.map((f) => f.split(" ").map(Number));
const visited = [...Array(h)].map((x) => Array.from(Array(w), () => 0));

function bfs(x, y) {
  visited[x][y] = 1;
  let cnt = 0;

  for (let i = 0; i < 4; i++) {
    const nX = x + dx[i];
    const nY = y + dy[i];

    if (nX < 0 || nX >= h || nY < 0 || nY >= w) continue;
    if (field[nX][nY] !== 0) continue;
    field[nX][nY] = 1;
    queue.enqueue([nX, nY]);
    cnt++;
  }
  return cnt;
}

function isAllRipe() {
  for (let i = 0; i < h; i++) {
    for (let j = 0; j < w; j++) {
      if (field[i][j] === 0) {
        return false;
      }
    }
  }
  return true;
}

const queue = new Queue();

for (let i = 0; i < h; i++) {
  for (let j = 0; j < w; j++) {
    if (field[i][j] === 1) {
      queue.enqueue([i, j]);
    }
  }
}

let cnt = 0;

while (queue.size()) {
  const size = queue.size();
  let plusCnt = 0;
  for (let i = 0; i < size; i++) {
    const [x, y] = queue.dequeue();
    plusCnt += bfs(x, y);
  }
  if (!plusCnt) break;
  cnt++;
  // console.log("cnt: ", cnt, "ì¶”ê°€ë¡œ ìµì€ í† ë§ˆí† : ", plusCnt);
}

console.log(isAllRipe() ? cnt : -1);
