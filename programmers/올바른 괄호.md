function solution(s) {
  const stack = [];

  for (let i of s.split("")) {
    if (i === "(") {
      stack.push(i);
    } else {
      if (stack.length === 0) stack.push(i);
      else if (stack[stack.length - 1] === "(") {
        stack.pop();
      }
    }
  }

  return stack.length ? false : true;
}
