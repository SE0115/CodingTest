function solution(num) {
  let answer = "";
  const otf = [4, 1, 2];
  while (num > 0) {
    let [quotient, remainder] = [parseInt(num / 3), num % 3];
    answer = otf[remainder] + answer;
    quotient = remainder === 0 ? quotient - 1 : quotient;
    num = quotient;
  }

  return answer;
}
