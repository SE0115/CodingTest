function solution(n) {
    let next = n + 1;

    while(count_1(n) !== count_1(next)){
      next += 1;
    }
    
    return next;
}

function count_1(num){
  let result ='';
  while(num){
    result = num % 2 +result;
    num = parseInt(num/2);
  }

  return result.split('').filter(x=>x==='1').length;
}
