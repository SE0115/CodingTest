function solution(number, k) {
  const answer = [];
  let deleteCnt = 0;

  for (let i of number) {
    if (answer.length === 0) answer.push(i);
    else {
      while (deleteCnt < k && i > answer[answer.length - 1]) {
        answer.pop();
        deleteCnt += 1;
      }
      answer.push(i);
    }
  }

  return answer.slice(0, number.length - k).join("");
}
