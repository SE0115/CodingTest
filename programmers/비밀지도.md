function solution(n, arr1, arr2) {
  var answer = [];
  for (let i = 0; i < n; i++) {
    let map = (arr1[i] | arr2[i]).toString(2);
    map = "0".repeat(n - map.length) + map;
    map = map.replace(/1/g, "#").replace(/0/g, " ");
    answer.push(map);
  }

  return answer;
}
