```js
function solution(s) {
  let answer = [];

  s = s
    .slice(2, s.length - 2)
    .split("},{")
    .map((x) => x.split(",").map(Number));
  s.sort((a, b) => a.length - b.length);

  for (i of s) {
    for (j of i) {
      if (!answer.includes(j)) answer.push(j);
    }
  }

  return answer;
}
```
