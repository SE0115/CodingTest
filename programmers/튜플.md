function solution(s) {
  let answer = [];
  let tuples = [];

  s = s.slice(1, s.length - 1).split("");
  let idx = 0;

  while (idx < s.length) {
    if (s[idx++] === "{") {
      let tmp = "";

      while (s[idx] !== "}") {
        tmp += s[idx++];
      }
      tuples.push(tmp.split(",").map(Number));
    }
  }
  tuples.sort((a, b) => a.length - b.length);

  for (i of tuples) {
    for (j of i) {
      if (!answer.includes(j)) answer.push(j);
    }
  }

  return answer;
}
