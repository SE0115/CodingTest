function solution(n) {
    
    n.sort(function(a,b) {
        return (''+a+b > ''+b+a) ? -1 : 1;
    });
    
    return ((n.every((x) => x===0)) ? '0' : n.join(''));
}

console.log(solution([0,0,0]));
console.log(solution([3, 30, 34, 5, 9]));
