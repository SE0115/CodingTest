function solution(p) {
  if (isRight(p)) {
    return p;
  } else {
    return makeRight(p);
  }
}

function isRight(str) {
  let stack = [];

  for (let i of str) {
    if (stack.length === 0) stack.push(i);
    else if (i === ")" && stack[stack.length - 1] === "(") stack.pop();
    else stack.push(i);
  }
  return stack.length ? false : true;
}

function splitUV(str) {
  let u = [];
  let v = [];

  if (str.length === 0) return "";

  for (let i of str) {
    if (u.length === 0) u.push(i);
    else {
      if (
        u.filter((x) => x === "(").length !== u.filter((x) => x === ")").length
      )
        u.push(i);
      else v.push(i);
    }
  }

  return [u.join(""), v.join("")];
}

function makeRight(str) {
  const splited = splitUV(str);
  if (splited === "") return "";

  let u = splited[0];
  let v = splited[1];

  if (isRight(u)) return u + makeRight(v);
  else {
    v = makeRight(v);

    if (u.length > 2) {
      u = u.slice(1, u.length - 1);
      let tmp = "";

      for (let i of u) {
        if (i === "(") tmp += ")";
        else tmp += "(";
      }
      u = tmp;
    } else u = "";

    return "(" + v + ")" + u;
  }
}
