const input =  require('fs').readFileSync('/dev/stdin').toString().trim().split('\n');

let cnt = 0;
input[1] = input[1].split(' ').map(x => Number(x));

for(let i=0; i<Number(input[0]); i++) {
    if(isPrime(input[1][i])){
        cnt+=1;
    }
}

console.log(cnt);

function isPrime(num){
    if(num === 1) {
        return false;
    }

    for(let i=2; i<=num**0.5; i++){
        if(num%i === 0){
            return false;
        }
    }
    
    return true;
}
